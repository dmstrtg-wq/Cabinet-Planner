// RENDER: FLOOR PLAN CANVAS

// ════════════════════════════

function renderCanvas() {

const canvas=document.getElementById('floor-plan');

const ctx=canvas.getContext('2d');

const r=activeRoom();

if(!r){ctx.clearRect(0,0,canvas.width,canvas.height);return;}

const scale=parseFloat(document.getElementById('scale-sel').value)||4;

const viewMode=document.getElementById('view-sel').value;

const roomW=Math.max(r.walls.north,r.walls.south,48);

const roomH=Math.max(r.walls.east,r.walls.west,48);

const PAD=68;

canvas.width =Math.max(roomW*scale+PAD*2,380);

canvas.height=Math.max(roomH*scale+PAD*2,280);

const RX=PAD, RY=PAD, RW=roomW*scale, RH=roomH*scale;


// Background + grid

ctx.fillStyle='#F8FAFC'; ctx.fillRect(0,0,canvas.width,canvas.height);

const GRID=12*scale;

ctx.strokeStyle='#E8EEF4'; ctx.lineWidth=0.5;

for(let x=RX;x<=RX+RW+1;x+=GRID){ctx.beginPath();ctx.moveTo(x,RY);ctx.lineTo(x,RY+RH);ctx.stroke();}

for(let y=RY;y<=RY+RH+1;y+=GRID){ctx.beginPath();ctx.moveTo(RX,y);ctx.lineTo(RX+RW,y);ctx.stroke();}

ctx.fillStyle='#FFF'; ctx.fillRect(RX,RY,RW,RH);


// Filter cabinets

let cabs=r.cabinets;

if(viewMode==='base') cabs=cabs.filter(c=>['base','sink','vanity','drawerBase','cornerBase','lazysusan'].includes(c.type));

if(viewMode==='wall') cabs=cabs.filter(c=>c.type==='wall');


// Draw cabinets per wall — wall cabs first (behind), then base cabs on top

// Corner/lazy susan cabinets get diagonal triangle treatment

function drawCabOnFloor(cab, wall) {

const cat=CATALOG[cab.type];

const cW=cab.width*scale, cD=cab.depth*scale;

const cabOff=(cab.offset||0)*scale;

const isActive=wall===state.activeWall;

const isCorner=cab.type==='cornerBase'||cab.type==='lazysusan';

let x,y,w,h;

if(wall==='north'){x=RX+cabOff;y=RY;w=cW;h=cD;}

else if(wall==='south'){x=RX+cabOff;y=RY+RH-cD;w=cW;h=cD;}

else if(wall==='west'){x=RX;y=RY+cabOff;w=cD;h=cW;}

else{x=RX+RW-cD;y=RY+cabOff;w=cD;h=cW;}


if(isCorner){

// Trapezoid: two sides flush to walls (right angles), one diagonal cut face

// North wall corner: top edge along north wall, left edge along west wall

// Shape has 4 points: corner-on-wall, end-along-wall, diagonal-cut, end-along-depth

ctx.globalAlpha=cab.type==='wall'?0.5:0.88;

ctx.fillStyle=cat.color;

ctx.beginPath();

if(wall==='north'){

// Sits in top-left of room: right angle at (x,y), extends right cW, down cD

// Diagonal cut from (x+cW, y) to (x, y+cD)

ctx.moveTo(x,y); // corner against wall

ctx.lineTo(x+cW,y); // along north wall

ctx.lineTo(x+cW,y+cD*0.35); // short front-right point

ctx.lineTo(x+cW*0.35,y+cD); // diagonal to front-left

ctx.lineTo(x,y+cD); // along west wall

} else if(wall==='south'){

ctx.moveTo(x,y+cD); // corner against south wall

ctx.lineTo(x+cW,y+cD); // along south wall

ctx.lineTo(x+cW,y+cD*0.65);

ctx.lineTo(x+cW*0.35,y);

ctx.lineTo(x,y);

} else if(wall==='west'){

ctx.moveTo(x,y); // corner against west wall

ctx.lineTo(x+cD,y); // along north wall

ctx.lineTo(x+cD,y+cW*0.35);

ctx.lineTo(x+cD*0.35,y+cW);

ctx.lineTo(x,y+cW); // down west wall

} else {

// East wall

ctx.moveTo(x+cD,y);

ctx.lineTo(x,y);

ctx.lineTo(x,y+cW*0.35);

ctx.lineTo(x+cD*0.65,y+cW);

ctx.lineTo(x+cD,y+cW);

}

ctx.closePath(); ctx.fill();

ctx.globalAlpha=1;

ctx.strokeStyle=isActive?cat.color:'rgba(0,0,0,0.2)';

ctx.lineWidth=isActive?2.5:1.5;

ctx.stroke();

// Dashed diagonal face line

ctx.strokeStyle='rgba(255,255,255,0.6)'; ctx.lineWidth=1; ctx.setLineDash([3,2]);

ctx.beginPath();

if(wall==='north'){ctx.moveTo(x+cW,y+cD*0.35);ctx.lineTo(x+cW*0.35,y+cD);}

else if(wall==='south'){ctx.moveTo(x+cW,y+cD*0.65);ctx.lineTo(x+cW*0.35,y);}

else if(wall==='west'){ctx.moveTo(x+cD,y+cW*0.35);ctx.lineTo(x+cD*0.35,y+cW);}

else{ctx.moveTo(x,y+cW*0.35);ctx.lineTo(x+cD*0.65,y+cW);}

ctx.stroke(); ctx.setLineDash([]);

} else {

const alpha=cab.type==='wall'?0.45:0.88;

ctx.globalAlpha=alpha;

ctx.fillStyle=cat.color;

ctx.fillRect(x,y,w,h);

ctx.globalAlpha=1;

ctx.strokeStyle=isActive?cat.color:'rgba(0,0,0,0.12)';

ctx.lineWidth=isActive?2:1;

ctx.strokeRect(x,y,w,h);

ctx.strokeStyle='rgba(255,255,255,0.45)'; ctx.lineWidth=0.5;

ctx.beginPath(); ctx.moveTo(x+2,y+h*0.45); ctx.lineTo(x+w-2,y+h*0.45); ctx.stroke();

}

const isDark=['#3B82F6','#1D4ED8','#06B6D4','#A855F7','#F59E0B','#F97316'].includes(cat.color);

ctx.fillStyle=isDark?'#fff':'#1e293b';

const fs=Math.max(8,Math.min(scale*2.0,11));

ctx.font=`bold ${fs}px sans-serif`;

ctx.textAlign='center'; ctx.textBaseline='middle';

if(isCorner){

// Label near the corner

if(wall==='north')ctx.fillText(`${cab.width}"`,x+cW*0.25,y+cD*0.35);

else if(wall==='south')ctx.fillText(`${cab.width}"`,x+cW*0.25,y+cD*0.65);

else if(wall==='west')ctx.fillText(`${cab.width}"`,x+cD*0.35,y+cW*0.25);

else ctx.fillText(`${cab.width}"`,x+cD*0.65,y+cW*0.25);

} else {

ctx.fillText(`${cab.width}"`,x+w/2,y+h/2);

}

}


['north','south','east','west'].forEach(wall=>{

// Wall cabs first (lower z-order)

cabs.filter(c=>c.wall===wall&&c.type==='wall').forEach(cab=>drawCabOnFloor(cab,wall));

// Base/tall cabs on top

cabs.filter(c=>c.wall===wall&&c.type!=='wall').forEach(cab=>drawCabOnFloor(cab,wall));

});


// Draw openings on floor plan using stored offset position

(r.openings||[]).forEach(op=>{

const wall=op.wall;

const oW=op.width*scale;

const opOffset=(op.offset||0)*scale;

let ox,oy,fw,fh;

if(wall==='north'){ox=RX+opOffset;oy=RY;fw=oW;fh=8;}

else if(wall==='south'){ox=RX+opOffset;oy=RY+RH-8;fw=oW;fh=8;}

else if(wall==='west'){ox=RX;oy=RY+opOffset;fw=8;fh=oW;}

else{ox=RX+RW-8;oy=RY+opOffset;fw=8;fh=oW;}


// White gap to show opening

ctx.fillStyle='#ffffff';

ctx.fillRect(ox,oy,fw,fh);


// Colored border

const opColors={'door':'#d97706','window':'#0284c7','sink-loc':'#0891b2','arch':'#7c3aed'};

ctx.strokeStyle=opColors[op.type]||'#d97706';

ctx.lineWidth=2; ctx.setLineDash([3,3]);

ctx.strokeRect(ox,oy,fw,fh);

ctx.setLineDash([]);


// Label icon

const icons={'door':'D','window':'W','sink-loc':'S','arch':'O'};

ctx.fillStyle=opColors[op.type]||'#d97706';

ctx.font='bold 8px sans-serif'; ctx.textAlign='center'; ctx.textBaseline='middle';

ctx.fillText(icons[op.type]||'?',ox+fw/2,oy+fh/2);

});


// Room outline

ctx.strokeStyle='#1E293B'; ctx.lineWidth=3; ctx.strokeRect(RX,RY,RW,RH);


// Dimension labels

ctx.fillStyle='#475569';

const df=Math.max(10,scale*2.3);

ctx.font=`600 ${df}px sans-serif`;

ctx.textAlign='center'; ctx.textBaseline='middle';

if(r.walls.north>0) ctx.fillText(fmtIn(r.walls.north),RX+RW/2,RY-22);

if(r.walls.south>0) ctx.fillText(fmtIn(r.walls.south),RX+RW/2,RY+RH+26);

ctx.save(); ctx.translate(RX-28,RY+RH/2); ctx.rotate(-Math.PI/2);

if(r.walls.west>0) ctx.fillText(fmtIn(r.walls.west),0,0);

ctx.restore();

ctx.save(); ctx.translate(RX+RW+28,RY+RH/2); ctx.rotate(Math.PI/2);

if(r.walls.east>0) ctx.fillText(fmtIn(r.walls.east),0,0);

ctx.restore();


// Compass

ctx.fillStyle='#94A3B8'; ctx.font='700 11px sans-serif';

ctx.textAlign='left'; ctx.textBaseline='bottom';

ctx.fillText('↑ N',RX+4,RY-28);


// Scale bar

const barPx=24*scale;

const bx=RX+RW-barPx-8, by=RY+RH+40;

ctx.strokeStyle='#475569'; ctx.lineWidth=2;

ctx.beginPath(); ctx.moveTo(bx,by); ctx.lineTo(bx+barPx,by); ctx.stroke();

ctx.beginPath(); ctx.moveTo(bx,by-4); ctx.lineTo(bx,by+4); ctx.stroke();

ctx.beginPath(); ctx.moveTo(bx+barPx,by-4); ctx.lineTo(bx+barPx,by+4); ctx.stroke();

ctx.fillStyle='#475569'; ctx.font='500 10px sans-serif';

ctx.textAlign='center'; ctx.textBaseline='top';

ctx.fillText('24"',bx+barPx/2,by+5);


renderSummary(r);

}

// RENDER: ELEVATION VIEW

// ════════════════════════════

function renderElevation() {

const canvas=document.getElementById('elevation-plan');

const ctx=canvas.getContext('2d');

const r=activeRoom();

if(!r){ctx.clearRect(0,0,canvas.width,canvas.height);return;}


const wall=state.elevWall;

const scale=parseFloat(document.getElementById('elev-scale-sel').value)||4;

const wallLength=r.walls[wall]||120;

const ceiling=r.ceilingHeight||96;

const p=activeProj();

const styleInfo=STYLES.find(s=>s.code===(p.style||'AW'))||STYLES[0];


const PAD=60;

const canvasW=Math.max(wallLength*scale+PAD*2,400);

const canvasH=Math.max(ceiling*scale+PAD*2,320);

canvas.width=canvasW; canvas.height=canvasH;


const WX=PAD, WY=PAD, WW=wallLength*scale, WH=ceiling*scale;


// Background

ctx.fillStyle='#F0F4F8'; ctx.fillRect(0,0,canvasW,canvasH);


// Wall surface

ctx.fillStyle='#FAFAFA'; ctx.fillRect(WX,WY,WW,WH);

ctx.strokeStyle='#CBD5E1'; ctx.lineWidth=0.5;

const gs=12*scale;

for(let x=WX;x<=WX+WW;x+=gs){ctx.beginPath();ctx.moveTo(x,WY);ctx.lineTo(x,WY+WH);ctx.stroke();}

for(let y=WY;y<=WY+WH;y+=gs){ctx.beginPath();ctx.moveTo(WX,y);ctx.lineTo(WX+WW,y);ctx.stroke();}


// Floor line

ctx.fillStyle='#CBD5E1'; ctx.fillRect(WX,WY+WH,WW,4);

// Counter line (36" from floor)

const counterY=WY+WH-36*scale;

ctx.strokeStyle='#94A3B8'; ctx.lineWidth=1; ctx.setLineDash([4,4]);

ctx.beginPath(); ctx.moveTo(WX,counterY); ctx.lineTo(WX+WW,counterY); ctx.stroke();

ctx.setLineDash([]);

ctx.fillStyle='#94A3B8'; ctx.font='500 9px sans-serif';

ctx.textAlign='left'; ctx.fillText('Counter Height 36"',WX+4,counterY-4);


// Get cabinets on this wall

const wallCabs=r.cabinets.filter(c=>c.wall===wall);

const wallOpenings=(r.openings||[]).filter(o=>o.wall===wall);


// Draw openings using stored offset position

wallOpenings.forEach(op=>{

const ox=WX+(op.offset||0)*scale;

const oW=op.width*scale;

const oH=op.height*scale;

const floorY=WY+WH;


if(op.type==='window'){

// Window sill at 42" from floor by default

const sillY=floorY-42*scale-oH;

ctx.fillStyle='#BAE6FD';

ctx.fillRect(ox+2,sillY,oW-4,oH);

async function sendAI() {

const input = document.getElementById('ai-input');

const msg = input.value.trim();

if (!msg) return;

input.value = '';


appendAIMessage('user', msg);

showTyping();


const btn = document.getElementById('ai-send-btn');

btn.disabled = true;


aiHistory.push({ role: 'user', content: msg });


// Build messages with fresh context each time

const contextMsg = { role: 'user', content: getProjectContext() + '\n\nUser request: ' + msg };

const messages = aiHistory.length <= 1

? [contextMsg]

: [...aiHistory.slice(0, -1), contextMsg];


try {

const resp = await fetch('https://api.anthropic.com/v1/messages', {

method: 'POST',

headers: { 'Content-Type': 'application/json', 'x-api-key': ANTHROPIC_API_KEY, 'anthropic-version': '2023-06-01', 'anthropic-dangerous-direct-browser-access': 'true' },

body: JSON.stringify({

model: 'claude-sonnet-4-20250514',

max_tokens: 1024,

system: buildSystemPrompt(),

messages

})

});


const data = await resp.json();

hideTyping();


if (!resp.ok) {

appendAIMessage('assistant', '⚠️ API error: ' + (data.error?.message || resp.status));

btn.disabled = false;

return;

}


const text = data.content?.[0]?.text || '';

aiHistory.push({ role: 'assistant', content: text });


// Extract display text (remove JSON block)

const displayText = text.replace(/```json[\s\S]*?```/g, '').trim();

appendAIMessage('assistant', displayText || '✓ Done!');


// Parse and execute command

const jsonMatch = text.match(/```json([\s\S]*?)```/);

if (jsonMatch) {

try {

const cmd = JSON.parse(jsonMatch[1].trim());

executeAICommand(cmd);

} catch(e) {

appendAIMessage('assistant', '⚠️ Could not parse the command. Please try rephrasing.');

}

}

} catch(e) {

hideTyping();

appendAIMessage('assistant', '⚠️ Connection error. Check your API key and try again.');

}

btn.disabled = false;

}


function executeAICommand(cmd) {

const r = activeRoom();

if (!r) return;


if (cmd.action === 'add_cabinet') {

const cat = CATALOG[cmd.type];

if (!cat) { appendAIMessage('assistant', '⚠️ Unknown cabinet type: ' + cmd.type); return; }

// Validate width

const validW = cat.widths.includes(Number(cmd.width));

const width = validW ? Number(cmd.width) : cat.widths.reduce((a,b) => Math.abs(b-cmd.width)<Math.abs(a-cmd.width)?b:a);

const height = cmd.height || cat.heights[0];

r.cabinets.push({

id: uid(), type: cmd.type, wall: cmd.wall||state.activeWall,

width, height, depth: cat.depth,

offset: cmd.offset || 0, note: cmd.note || ''

});

persist(); renderCanvas(); renderCabinetList();

if (state.viewMode === 'elevation') renderElevation();

}


else if (cmd.action === 'remove_cabinet') {

const before = r.cabinets.length;

r.cabinets = r.cabinets.filter(c => {

if (cmd.wall && c.wall !== cmd.wall) return true;

if (cmd.type && c.type !== cmd.type) return true;

if (cmd.width && c.width !== Number(cmd.width)) return true;

return false;

});

const removed = before - r.cabinets.length;

persist(); renderCanvas(); renderCabinetList();

if (!removed) appendAIMessage('assistant', 'No matching cabinets found to remove.');

}


else if (cmd.action === 'move_cabinet') {

const cab = r.cabinets.find(c =>

c.wall === cmd.wall &&

c.type === cmd.type &&

c.width === Number(cmd.width) &&

(c.offset||0) === Number(cmd.identify_offset)

);

if (cab) { cab.offset = Number(cmd.new_offset); persist(); renderCanvas(); renderCabinetList(); }

else appendAIMessage('assistant', "Couldn't find that cabinet to move. Try being more specific.");

}


else if (cmd.action === 'clear_wall') {

r.cabinets = r.cabinets.filter(c => c.wall !== cmd.wall);

persist(); renderCanvas(); renderCabinetList();

}


else if (cmd.action === 'add_opening') {

if (!r.openings) r.openings = [];

r.openings.push({

id: uid(), type: cmd.type || 'door',

wall: cmd.wall || state.activeWall,

width: Number(cmd.width) || 36,

height: Number(cmd.height) || 80,

offset: Number(cmd.offset) || 0

});

persist(); renderCanvas(); renderCabinetList();

}

}
